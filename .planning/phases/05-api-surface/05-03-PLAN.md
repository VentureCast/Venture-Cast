    1. ammSchemas.js — ADD (keep existing exports):
       createMarketSchema = Joi.object({ streamerId: objectId.required(),
         P0_cents: Joi.number().integer().min(1).optional(), k_num: Joi.number().integer().min(1).optional(),
         k_den: Joi.number().integer().min(1).optional(), tier: Joi.string().valid('1','2','3').optional(),
         spreadBps: Joi.number().integer().min(0).optional(), feeBps: Joi.number().integer().min(0).optional(),
         reserveFloorCents: Joi.number().integer().min(1).required() });
       patchMarketSchema = Joi.object({ status: Joi.string().valid('active','paused').optional(),
         tier: Joi.string().valid('1','2','3').optional(),
         spreadBps: Joi.number().integer().min(0).optional(), feeBps: Joi.number().integer().min(0).optional() })
         .min(1);
       Export them alongside marketIdParam/quoteSchema/orderSchema.
    2. controllers/ammController.js — ADD handlers (keep existing exports). Require openMarket
       (genesisService), Market, RiskEvent, LedgerEntry, LedgerAccount, AdminAction, logger.
       Standard try/catch: on err.statusCode -> res.status(err.statusCode).json({ error: err.message,
       ...err.details }); else logger.error + 500.
       - adminCreateMarket(req,res): const { market, marketState } = await openMarket(req.body);
         await AdminAction.create({ adminId: req.userId, action: 'create_market',
         target: market._id.toString(), after: { tier: market.tier } }); res.status(201).json({ market, marketState }).
       - adminPatchMarket(req,res): const m = await Market.findById(req.params.id); if !m -> 404
         { error: 'market not found' }. const before = { status:m.status, tier:m.tier, spreadBps:m.spreadBps,
         feeBps:m.feeBps }. Apply provided fields from req.body (status/tier/spreadBps/feeBps) to m; m.save().
         action = req.body.status==='paused' ? 'pause_market' : req.body.status==='active' ? 'resume_market'
         : 'set_tier'. await AdminAction.create({ adminId:req.userId, action, target:m._id.toString(),
         before, after:{ status:m.status, tier:m.tier } }); res.json({ market: m }).
       - adminListRiskEvents(req,res): const limit = Math.min(parseInt(req.query.limit)||50, 200);
         const events = await RiskEvent.find().sort({ createdAt:-1 }).limit(limit).lean(); res.json({ events }).
       - adminReconcile(req,res): build per-account expected sums from LedgerEntry:
         const entries = await LedgerEntry.find().lean(); const expected = {}; let total = 0;
         for each e: expected[e.accountKey]=(expected[e.accountKey]||0)+e.delta; total += e.delta.
         const accounts = await LedgerAccount.find().lean(); compare each account.balance to
         expected[account.accountKey] (default 0); collect mismatches. balanced = (total===0) &&
         mismatches.length===0. res.json({ balanced, totalCents: total, mismatches }). (totalCents
         MUST be 0 for a healthy ledger; per-account projection MUST equal the summed entries.)
       Export adminCreateMarket, adminPatchMarket, adminListRiskEvents, adminReconcile.
  </action>
